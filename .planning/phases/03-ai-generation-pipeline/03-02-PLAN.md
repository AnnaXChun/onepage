    import lombok.RequiredArgsConstructor;
    import lombok.extern.slf4j.Slf4j;
    import org.springframework.amqp.rabbit.core.RabbitTemplate;
    import org.springframework.stereotype.Component;

    @Component
    @RequiredArgsConstructor
    @Slf4j
    public class GenerationMessageProducer {

        private final RabbitTemplate rabbitTemplate;

        public void sendGenerationRequest(Long blogId, String imageUrl, String description,
                                          String[] colorPalette, String dominantColor) {
            log.info("Sending generation request to queue for blogId: {}", blogId);
            rabbitTemplate.convertAndSend(
                "blog.generate.exchange",
                "blog.generate",
                new GenerationMessage(blogId, imageUrl, description, colorPalette, dominantColor)
            );
        }

        public record GenerationMessage(
            Long blogId,
            String imageUrl,
            String description,
            String[] colorPalette,
            String dominantColor
        ) {}
    }

    package com.onepage.messaging;

    import com.onepage.dto.GenerationRequest;
    import com.onepage.dto.GenerationResult;
    import com.onepage.service.AIGenerationService;
    import lombok.RequiredArgsConstructor;
    import lombok.extern.slf4j.Slf4j;
    import org.springframework.amqp.rabbit.annotation.RabbitListener;
    import org.springframework.messaging.simp.SimpMessagingTemplate;
    import org.springframework.stereotype.Component;

    @Component
    @RequiredArgsConstructor
    @Slf4j
    public class GenerationMessageConsumer {

        private final AIGenerationService generationService;
        private final SimpMessagingTemplate messagingTemplate;

        @RabbitListener(queues = "blog.generate.queue")
        public void handleGenerationRequest(GenerationMessageProducer.GenerationMessage message) {
            Long blogId = message.blogId();
            try {
                // AI-05: Send progress - starting
                sendProgress(blogId, "STARTING", 0);

                // Build request and generate
                GenerationRequest request = GenerationRequest.builder()
                    .blogId(blogId)
                    .imageUrl(message.imageUrl())
                    .description(message.description())
                    .colorPalette(message.colorPalette())
                    .dominantColor(message.dominantColor())
                    .build();

                sendProgress(blogId, "GENERATING", 25);
                GenerationResult result = generationService.generate(request);

                sendProgress(blogId, "ASSEMBLING_BLOCKS", 75);
                // Save blocks to blog...

                sendProgress(blogId, "COMPLETED", 100);
                log.info("Generation completed for blogId: {}", blogId);

            } catch (Exception e) {
                log.error("Generation failed for blogId: {}", blogId, e);
                sendProgress(blogId, "FAILED", 0);
            }
        }

        private void sendProgress(Long blogId, String stage, int percent) {
            // AI-06: Progress via WebSocket
            messagingTemplate.convertAndSend(
                "/topic/progress/" + blogId,
                new ProgressUpdate(stage, percent)
            );
        }

        public record ProgressUpdate(String stage, int percent) {}
    }

    package com.onepage.controller;

    import com.onepage.messaging.GenerationMessageProducer;
    import com.onepage.dto.GenerationRequest;
    import lombok.RequiredArgsConstructor;
    import org.springframework.http.ResponseEntity;
    import org.springframework.web.bind.annotation.*;

    @RestController
    @RequestMapping("/api/ai")
    @RequiredArgsConstructor
    public class AIGenerationController {

        private final GenerationMessageProducer messageProducer;

        @PostMapping("/generate")
        public ResponseEntity<?> startGeneration(@RequestBody GenerationRequest request) {
            // AI-01: Trigger generation with image + description
            // Validate required fields
            if (request.getBlogId() == null) {
                return ResponseEntity.badRequest().body("blogId required");
            }
            if (request.getImageUrl() == null || request.getDescription() == null) {
                return ResponseEntity.badRequest().body("imageUrl and description required");
            }

            // AI-05: Send to async queue - return immediately
            messageProducer.sendGenerationRequest(
                request.getBlogId(),
                request.getImageUrl(),
                request.getDescription(),
                request.getColorPalette(),
                request.getDominantColor()
            );

            return ResponseEntity.accepted().body("Generation started");
        }
    }
