    @Service
    @RequiredArgsConstructor
    public class SiteService extends ServiceImpl<BlogMapper, Blog> {
        private final BlogMapper blogMapper;
        private final UserService userService;

        public Blog getPublishedBlogByUsername(String username) {
            User user = userService.findByUsername(username);
            if (user == null) {
                throw BusinessException.notFound("User not found");
            }
            Blog blog = this.lambdaQuery()
                .eq(Blog::getUserId, user.getId())
                .eq(Blog::getStatus, 1)  // published
                .orderByDesc(Blog::getPublishTime)
                .one();
            if (blog == null) {
                throw BusinessException.notFound("No published blog found for user");
            }
            return blog;
        }
    }

<task type="auto">
  <name>Task 4: Create SubdomainFilter for subdomain extraction and routing</name>
  <files>backend/src/main/java/com/onepage/filter/SubdomainFilter.java</files>
  <action>
    Create SubdomainFilter.java that:
    1. Extracts subdomain from Host header (e.g., "john.localhost:8080" -> "john")
    2. In dev mode (environment=dev), checks if Host matches pattern *.localhost:port
    3. In prod mode (environment=prod), checks if Host matches pattern *.vibe.com
    4. If subdomain detected and looks like a username, forwards to /host/{subdomain}
    5. Otherwise passes request through unchanged

    ```java
    @Component
    @Order(Ordered.HIGHEST_PRECEDENCE)
    public class SubdomainFilter extends OncePerRequestFilter {
        private final HostingConfig hostingConfig;

        @Override
        protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
                throws ServletException, IOException {
            String host = request.getHeader("Host");
            if (host == null) {
                filterChain.doFilter(request, response);
                return;
            }

            String subdomain = extractSubdomain(host);
            if (subdomain != null && isValidUsername(subdomain)) {
                // Forward to SiteController
                request.getRequestDispatcher("/host/" + subdomain).forward(request, response);
                return;
            }

            filterChain.doFilter(request, response);
        }

        private String extractSubdomain(String host) {
            // Handle localhost:port format
            if (host.contains("localhost") || host.contains("127.0.0.1")) {
                // For dev: john.localhost:8080 -> extract "john"
                String[] parts = host.split("\\.");
                if (parts.length >= 3) {
                    return parts[0];  // First part before localhost
                }
            }
            // Handle production *.vibe.com format
            if (host.contains("vibe.com")) {
                String[] parts = host.split("\\.");
                if (parts.length >= 2) {
                    return parts[0];  // First part before vibe.com
                }
            }
            return null;
        }

        private boolean isValidUsername(String subdomain) {
            // Username should be alphanumeric, 3-30 chars
            return subdomain != null && subdomain.matches("^[a-zA-Z0-9_]{3,30}$");
        }
    }
    ```
