    ```typescript
    export const resendVerification = async (emailOrUsername: string) => {
      const response = await axios.post('/api/user/resend-verification', { email: emailOrUsername });
      return response.data;
    };

    // Verify email with token (for verification page)
    export const verifyEmail = async (token: string) => {
      const response = await axios.post('/api/user/verify-email', null, { params: { token } });
      return response.data;
    };

    // Update user email
    export const updateEmail = async (email: string) => {
      const response = await axios.put('/api/user/email', { email });
      return response.data;
    };
    ```

    Also update getUserInfo to include full user data from backend:
    ```typescript
    export const getUserInfo = async () => {
      const response = await axios.get('/api/user/info');
      // Update localStorage with full user data
      if (response.data.code === 200) {
        localStorage.setItem('user', JSON.stringify(response.data.data));
      }
      return response.data;
    };
    ```
